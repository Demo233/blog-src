```java
    public void dfs(int i) {
	// 输出访问的节点
        System.out.print(vertexs.get(i).toString() + "=>");
        visited[i] = true;

        //获取i的第一个邻接点
        int w = firstNeighbor(i);
        while (w != -1) {
	    // 判断是否访问过
            if (!visited[w]) {
		// 没有访问递归
                dfs(w);
            }
	    // 找下个邻接点
            w = nextNeighbor(i, w);
        }

    }

    // 深度优先遍历算法
    public void dfs() {
	// 这里对上面dfs(i)方法进行重载，dfs(i)是找一个点的所有邻接点，这里套一层意思就是找所有点的邻接点
        for (int i = 0; i < vertexs.size(); i++) {
            if (!visited[i]) {
                dfs(i);
            }
        }
    }
```
